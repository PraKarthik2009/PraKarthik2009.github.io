  <script>
    fetch("drive.json")
      .then(res => res.json())
      .then(data => {
        const gallery = document.getElementById("pdf-gallery");
        if (!data.length) {
          gallery.innerHTML = '<span style="color:#888;">No PDFs found.</span>';
          return;
        }
        data.forEach(file => {
          const card = document.createElement("div");
          card.className = "pdf-card";
          const title = document.createElement("div");
          title.className = "pdf-title";
          title.textContent = file.name || "Untitled PDF";
          card.appendChild(title);
          const iframe = document.createElement("iframe");
          iframe.className = "pdf-frame";
          iframe.src = `https://docs.google.com/gview?url=https://drive.google.com/uc?id=${file.id}&embedded=true`;
          iframe.frameBorder = "0";
          iframe.allowFullscreen = true;
          card.appendChild(iframe);
          gallery.appendChild(card);
        });
      })
      .catch(err => {
        document.getElementById("pdf-gallery").innerText = "Failed to load PDFs.";
        console.error(err);
      });
  </script>
